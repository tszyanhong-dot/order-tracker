<script>
  const ORDER_KEY = 'order_tracker_v2_orders';
  const PRODUCT_KEY = 'order_tracker_v2_products';
  const VARIANT_KEY = 'order_tracker_v2_variants';

  let orders = [];
  let products = [];
  let variantHistory = [];
  let pendingPhoto = null;
  let editingOrderId = null;

  // ---------- IndexedDB 基礎操作封裝 ----------
  const DB_NAME = 'OrderTrackerDB';
  const DB_VERSION = 1;
  let db = null;

  function initDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION);
      request.onupgradeneeded = function(e) {
        const database = e.target.result;
        if (!database.objectStoreNames.contains('products')) {
          database.createObjectStore('products', { keyPath: 'id' });
        }
        if (!database.objectStoreNames.contains('orders')) {
          database.createObjectStore('orders', { keyPath: 'id' });
        }
        if (!database.objectStoreNames.contains('variants')) {
          database.createObjectStore('variants', { keyPath: 'id' });
        }
      };
      request.onsuccess = function(e) {
        db = e.target.result;
        resolve();
      };
      request.onerror = function(e) {
        reject(e.target.error);
      };
    });
  }

  function dbGetStore(storeName, mode = 'readonly') {
    return db.transaction(storeName, mode).objectStore(storeName);
  }

  function dbGetAll(storeName) {
    return new Promise((resolve, reject) => {
      const store = dbGetStore(storeName);
      const req = store.getAll();
      req.onsuccess = () => resolve(req.result);
      req.onerror = () => reject(req.target.error);
    });
  }

  function dbWrite(storeName, item) {
    return new Promise((resolve, reject) => {
      const store = dbGetStore(storeName, 'readwrite');
      const req = store.put(item);
      req.onsuccess = () => resolve();
      req.onerror = () => reject(req.target.error);
    });
  }

  function dbDelete(storeName, id) {
    return new Promise((resolve, reject) => {
      const store = dbGetStore(storeName, 'readwrite');
      const req = store.delete(id);
      req.onsuccess = () => resolve();
      req.onerror = () => reject(req.target.error);
    });
  }

  function dbClearStore(storeName) {
    return new Promise((resolve, reject) => {
      const store = dbGetStore(storeName, 'readwrite');
      const req = store.clear();
      req.onsuccess = () => resolve();
      req.onerror = () => reject(req.target.error);
    });
  }

  // ---------- 數據載入與無縫遷移 (LocalStorage -> IndexedDB) ----------
  async function loadData() {
    await initDB();
    
    // 檢查是否有舊 localStorage 資料需要遷移
    let localOrders = [];
    let localProducts = [];
    let localVariants = [];
    try { localOrders = JSON.parse(localStorage.getItem(ORDER_KEY)) || []; } catch(e){}
    try { localProducts = JSON.parse(localStorage.getItem(PRODUCT_KEY)) || []; } catch(e){}
    try { localVariants = JSON.parse(localStorage.getItem(VARIANT_KEY)) || []; } catch(e){}

    if (localOrders.length > 0 || localProducts.length > 0) {
      // 遷移至 IndexedDB
      for (const o of localOrders) {
        if (!o.status) o.status = '未採購'; // 補回狀態預設值
        await dbWrite('orders', o);
      }
      for (const p of localProducts) {
        await dbWrite('products', p);
      }
      for (const v of localVariants) {
        await dbWrite('variants', { id: v, name: v });
      }
      // 遷移後清除舊儲存空間
      localStorage.removeItem(ORDER_KEY);
      localStorage.removeItem(PRODUCT_KEY);
      localStorage.removeItem(VARIANT_KEY);
      console.log('LocalStorage 舊資料已成功無縫遷移至 IndexedDB！');
    }

    // 從 IndexedDB 載入現有數據
    orders = await dbGetAll('orders');
    products = await dbGetAll('products');
    const vRaw = await dbGetAll('variants');
    variantHistory = vRaw.map(v => v.name);

    refreshProductSelect();
  }

  function escapeHtml(s) {
    return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

  function switchTab(name) {
    ['add','products','list','summary'].forEach(n => {
      document.getElementById('view-' + n).style.display = (n === name) ? 'block' : 'none';
      document.getElementById('tab-' + n).classList.toggle('active', n === name);
    });
    if (name === 'products') renderProductLibrary();
    if (name === 'list') renderList();
    if (name === 'summary') renderSummary();
    if (name === 'add') refreshProductSelect();
  }

  // ---------- 客戶編號格式校對與自動大寫 ----------
  function validateCustomerFormat(val) {
    if (!val) return { valid: false, msg: '請輸入客戶編號' };
    const regex = /^[WL]\d{7}$|^\d{8}$/;
    if (regex.test(val)) {
      return { valid: true, msg: '✓ 格式正確' };
    }
    return { valid: false, msg: '✗ 格式不符 (W/L+7位數字 或 8位電話)' };
  }

  function setupCustomerInputValidation(inputId, hintId) {
    const input = document.getElementById(inputId);
    const hint = document.getElementById(hintId);
    
    input.addEventListener('input', function() {
      let val = this.value;
      // 若首字母是 w/l 則自動轉大寫
      if (/^[wlWL]/i.test(val)) {
        val = val.toUpperCase();
        this.value = val;
      }
      
      if (!val) {
        hint.textContent = '格式：W1234567 / L1234567 / 8位數字電話';
        hint.style.color = 'var(--text-muted)';
        return;
      }
      
      const result = validateCustomerFormat(val);
      hint.textContent = result.msg;
      hint.style.color = result.valid ? 'var(--profit)' : 'var(--danger)';
    });
  }

  // ---------- 產品庫 ----------
  document.getElementById('new-product-photo').addEventListener('change', function(e) {
    const file = e.target.files[0];
    if (!file) { pendingPhoto = null; return; }
    const reader = new FileReader();
    reader.onload = function(ev) {
      const img = new Image();
      img.onload = function() {
        const maxW = 500;
        const scale = Math.min(1, maxW / img.width);
        const canvas = document.createElement('canvas');
        canvas.width = img.width * scale;
        canvas.height = img.height * scale;
        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
        pendingPhoto = canvas.toDataURL('image/jpeg', 0.7);
        const preview = document.getElementById('new-product-preview');
        preview.src = pendingPhoto;
        preview.style.display = 'block';
      };
      img.src = ev.target.result;
    };
    reader.readAsDataURL(file);
  });

  async function addProduct() {
    const name = document.getElementById('new-product-name').value.trim();
    if (!name) { alert('請輸入產品名稱'); return; }
    
    const newProd = {
      id: Date.now(),
      name: name,
      photo: pendingPhoto || ''
    };
    
    await dbWrite('products', newProd);
    products.push(newProd);

    document.getElementById('new-product-name').value = '';
    document.getElementById('new-product-photo').value = '';
    document.getElementById('new-product-preview').style.display = 'none';
    pendingPhoto = null;
    renderProductLibrary();
    alert('產品已加入產品庫，而家可以去「新增訂單」揀返佢');
  }

  async function deleteProduct(id) {
    if (!confirm('刪除呢個產品？已經記錄低嘅訂單唔會受影響。')) return;
    await dbDelete('products', id);
    products = products.filter(p => p.id !== id);
    renderProductLibrary();
  }

  function renderProductLibrary() {

  function onProductSelect() {
    const id = document.getElementById('product-select').value;
    const preview = document.getElementById('product-preview');
    if (!id) { preview.style.display = 'none'; return; }
    const p = products.find(x => String(x.id) === String(id));
    if (!p) { preview.style.display = 'none'; return; }
    document.getElementById('preview-img').src = p.photo || '';
    document.getElementById('preview-name').textContent = p.name;
    preview.style.display = 'flex';
  }

  async function addOrder() {
    const customer = document.getElementById('customer').value.trim();
    const productId = document.getElementById('product-select').value;
    const variant = document.getElementById('variant').value.trim();
    const qty = parseInt(document.getElementById('qty').value) || 0;
    const jpyPrice = document.getElementById('jpyPrice').value;
    const hkdPrice = document.getElementById('hkdPrice').value;

    if (!customer || !productId || qty <= 0) {
      alert('請輸入客戶編號、揀返產品，同埋數量');
      return;
    }

    const validation = validateCustomerFormat(customer);
    if (!validation.valid) {
      if (!confirm('客戶編號格式不符標準，確定要繼續加入嗎？')) {
        return;
      }
    }

    const product = products.find(p => String(p.id) === String(productId));

    const newOrder = {
      id: Date.now(),
      customer,
      productId: productId,
      productName: product ? product.name : '（已刪除產品）',
      productPhoto: product ? product.photo : '',
      variant,
      qty,
      jpyPrice: jpyPrice ? parseFloat(jpyPrice) : null,
      hkdPrice: hkdPrice ? parseFloat(hkdPrice) : null,
      status: '未採購', // 預設為未採購
      time: new Date().toLocaleString('zh-HK', { hour12: false })
    };

    await dbWrite('orders', newOrder);
    orders.push(newOrder);

    if (variant && !variantHistory.includes(variant)) {
      variantHistory.push(variant);
      await dbWrite('variants', { id: variant, name: variant });
    }

    refreshProductSelect();

    // 恢復預設
    document.getElementById('customer').value = '';
    document.getElementById('customer-validation').textContent = '格式：W1234567 / L1234567 / 8位數字電話';
    document.getElementById('customer-validation').style.color = 'var(--text-muted)';
    document.getElementById('qty').value = 1;
    document.getElementById('variant').value = '';
    document.getElementById('jpyPrice').value = '';
    document.getElementById('hkdPrice').value = '';
    alert('訂單加入成功！');
  }

  async function deleteOrder(id) {
    if (!confirm('確定刪除呢個訂單記錄？')) return;
    await dbDelete('orders', id);
    orders = orders.filter(o => o.id !== id);
    renderList();
    renderSummary();
    refreshProductSelect();
  }

  // ---------- 編輯訂單彈窗邏輯 ----------
  function openEditModal(orderId) {
    const o = orders.find(x => x.id === orderId);
    if (!o) return;
    editingOrderId = orderId;
    
    document.getElementById('edit-customer').value = o.customer;
    document.getElementById('edit-variant').value = o.variant || '';
    document.getElementById('edit-qty').value = o.qty;
    document.getElementById('edit-jpyPrice').value = o.jpyPrice !== null ? o.jpyPrice : '';
    document.getElementById('edit-hkdPrice').value = o.hkdPrice !== null ? o.hkdPrice : '';
    document.getElementById('edit-status').value = o.status || '未採購';

    const validation = validateCustomerFormat(o.customer);
    const hint = document.getElementById('edit-customer-validation');
    hint.textContent = validation.msg;
    hint.style.color = validation.valid ? 'var(--profit)' : 'var(--danger)';

    document.getElementById('edit-modal').classList.add('active');
  }

  function closeEditModal() {
    document.getElementById('edit-modal').classList.remove('active');
    editingOrderId = null;
  }

  async function saveEditOrder() {
    if (!editingOrderId) return;
    const oIndex = orders.findIndex(x => x.id === editingOrderId);
    if (oIndex === -1) return;

    const customer = document.getElementById('edit-customer').value.trim();
    const variant = document.getElementById('edit-variant').value.trim();
    const qty = parseInt(document.getElementById('edit-qty').value) || 0;
    const jpyPrice = document.getElementById('edit-jpyPrice').value;
    const hkdPrice = document.getElementById('edit-hkdPrice').value;
    const status = document.getElementById('edit-status').value;

    if (!customer || qty <= 0) {
      alert('請填寫客戶編號同數量');
      return;
    }

    const updatedOrder = {
      ...orders[oIndex],
      customer,
      variant,
      qty,
      jpyPrice: jpyPrice ? parseFloat(jpyPrice) : null,
      hkdPrice: hkdPrice ? parseFloat(hkdPrice) : null,
      status
    };

    await dbWrite('orders', updatedOrder);
    orders[oIndex] = updatedOrder;

    if (variant && !variantHistory.includes(variant)) {
      variantHistory.push(variant);
      await dbWrite('variants', { id: variant, name: variant });
    }

    closeEditModal();
    renderList();
    renderSummary();
    refreshProductSelect();
    alert('修改已儲存！');
  }

  // ---------- 訂單清單 (搜尋與篩選) ----------
  function getStatusClass(status) {
    if (status === '已採購') return 'status-sourced';
    if (status === '斷貨') return 'status-outofstock';
    if (status === '已交件') return 'status-delivered';
    return 'status-pending';
  }

  function renderList() {
    const container = document.getElementById('order-list');
    
    const searchVal = document.getElementById('search-list').value.toLowerCase().trim();
    const statusFilter = document.getElementById('filter-status').value;

    // 進行過濾
    const filteredOrders = orders.filter(o => {
      const matchSearch = 
        o.customer.toLowerCase().includes(searchVal) ||
        o.productName.toLowerCase().includes(searchVal) ||
        (o.variant && o.variant.toLowerCase().includes(searchVal));
      
      const matchStatus = (statusFilter === 'all') || (o.status === statusFilter);
      
      return matchSearch && matchStatus;
    });

    document.getElementById('list-count').textContent = filteredOrders.length;
    
    if (filteredOrders.length === 0) {
      container.innerHTML = '<div class="empty">無符合搜尋或篩選條件嘅記錄</div>';
      return;
    }

    const sorted = [...filteredOrders].reverse();
    container.innerHTML = sorted.map(o => `
      <div class="order-row">
        ${o.productPhoto ? `<img class="thumb" src="${o.productPhoto}">` : `<div class="thumb"></div>`}
        <div class="order-main">
          <div class="order-customer">
            ${escapeHtml(o.customer)}
            <span class="status-badge ${getStatusClass(o.status)}">${escapeHtml(o.status || '未採購')}</span>
          </div>
          <div class="order-product">${escapeHtml(o.productName)}</div>
          ${o.variant ? `<span class="order-variant">${escapeHtml(o.variant)}</span>` : ''}
          ${(o.jpyPrice || o.hkdPrice) ? `<div class="order-price">${o.jpyPrice ? '¥'+o.jpyPrice : ''} ${o.hkdPrice ? '$'+o.hkdPrice : ''}</div>` : ''}
        </div>
        <div class="order-qty">x${o.qty}</div>
        <button class="del-btn" onclick="openEditModal(${o.id})" style="font-size: 13px; color: var(--accent); font-weight:600; padding-right:10px;">編輯</button>
        <button class="del-btn" onclick="deleteOrder(${o.id})">✕</button>
      </div>
    `).join('');
  }

  // ---------- 統計 (自動排除斷貨訂單) ----------
  function renderSummary() {
    const byProduct = {};
    // 只統計非「斷貨」嘅訂單
    const activeOrders = orders.filter(o => o.status !== '斷貨');

    activeOrders.forEach(o => {
      const key = o.productId || o.productName;
      if (!byProduct[key]) {
        byProduct[key] = { name: o.productName, photo: o.productPhoto, total: 0, variants: {} };
      }
      byProduct[key].total += o.qty;
      const vKey = o.variant || '（未分款式）';
      byProduct[key].variants[vKey] = (byProduct[key].variants[vKey] || 0) + o.qty;
    });

    const productEl = document.getElementById('summary-product');
    const entries = Object.values(byProduct).sort((a,b) => b.total - a.total);

    if (entries.length === 0) {
      productEl.innerHTML = '<div class="empty">未有採購記錄（或所有訂單皆為斷貨）</div>';
    } else {
      productEl.innerHTML = entries.map(p => {
        const variantKeys = Object.keys(p.variants);
        const showVariants = variantKeys.length > 1 || (variantKeys.length === 1 && variantKeys[0] !== '（未分款式）');
        const variantHtml = showVariants ? variantKeys.map(v => `
          <div class="variant-row"><span class="v-name">${escapeHtml(v)}</span><span>${p.variants[v]}</span></div>
        `).join('') : '';
        return `
          <div class="summary-block">
            <div class="summary-head">
              ${p.photo ? `<img class="thumb" src="${p.photo}">` : `<div class="thumb"></div>`}
              <span style="font-size:14px; font-weight:600;">${escapeHtml(p.name)}</span>
              <span class="summary-total">${p.total}</span>
            </div>
            ${variantHtml}
          </div>
        `;
      }).join('');
    }

    const byCustomer = {};
    activeOrders.forEach(o => { byCustomer[o.customer] = (byCustomer[o.customer] || 0) + o.qty; });
    const customerEl = document.getElementById('summary-customer');
    const cEntries = Object.entries(byCustomer).sort((a,b) => b[1]-a[1]);
    customerEl.innerHTML = cEntries.length ? cEntries.map(([name, qty]) => `
      <div class="variant-row" style="padding-left:0;"><span class="v-name">${escapeHtml(name)}</span><span>${qty} 件</span></div>
    `).join('') : '<div class="empty">未有採購記錄</div>';
  }

  // ---------- 匯出 & 備份 (CSV & JSON) ----------
  function exportCSV() {
    if (orders.length === 0) { alert('未有記錄可以匯出'); return; }
    let csv = '\uFEFF客戶編號,產品名稱,款式,數量,日圓貨價,港幣貨價,代購狀態,時間\n';
    orders.forEach(o => {
      csv += `${o.customer},"${o.productName}","${o.variant || ''}",${o.qty},${o.jpyPrice || ''},${o.hkdPrice || ''},"${o.status || '未採購'}",${o.time}\n`;
    });
    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = '訂單記錄_' + new Date().toISOString().slice(0,10) + '.csv';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }

  // 還原完整備份
  async function importJSON(event) {
    const file = event.target.files[0];
    if (!file) return;

    if (!confirm('匯入備份會取代目前手機內嘅所有訂單同埋商品圖庫。確定要覆蓋並還原？')) {
      event.target.value = '';
      return;
    }

    const reader = new FileReader();
    reader.onload = async function(e) {
      try {
        const data = JSON.parse(e.target.result);
        if (!data.orders || !data.products) {
          alert('匯入失敗：檔案格式不正確，找不到訂單或商品數據！');
          return;
        }

        // 1. 清空 IndexedDB 舊資料
        await dbClearStore('orders');
        await dbClearStore('products');
        await dbClearStore('variants');

        // 2. 寫入新備份數據
        for (const o of data.orders) {
          await dbWrite('orders', o);
        }
        for (const p of data.products) {
          await dbWrite('products', p);
        }
        if (data.variants) {
          for (const v of data.variants) {
            await dbWrite('variants', { id: v, name: v });
          }
        }

        alert('還原成功！網頁即將重新載入。');
        window.location.reload();
      } catch (err) {
        alert('解析備份檔案時發生錯誤：' + err.message);
      }
    };
    reader.readAsText(file);
  }

  async function clearAll() {
    if (confirm('確定要清空所有訂單記錄嗎？產品庫唔會被刪走。呢個動作唔可以復原。建議先匯出完整備份 (.json)')) {
      await dbClearStore('orders');
      orders = [];
      renderList();
      renderSummary();
      refreshProductSelect();
    }
  }
